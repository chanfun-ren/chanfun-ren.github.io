class Spinlock {
 private:
  std::atomic_flag flag = ATOMIC_FLAG_INIT;

 public:
  void lock() {
    while (flag.test_and_set())
      ;
  }
  void unlock() { flag.clear(); }
};

void WorkOnResource() {
  spin.lock();
  spin.unlock();
}

int main(int argc, char const *argv[]) {
  std::thread t(WorkOnResource);
  std::thread t1(WorkOnResource);

  t.join();
  t1.join();
  return 0;
}
